project(Bindings.Sofa.Simulation)

)

)

        SOURCES      ${SOURCE_FILES}
        HEADERS      ${HEADER_FILES}
        DEPENDS      SofaCore SofaSimulationCore SofaSimulationGraph SofaPython3::Plugin
)
